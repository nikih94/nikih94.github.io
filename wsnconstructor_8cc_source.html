243 void
244 WsnConstructor::ProcessPathString ()
245 {
246  //count the number of paths
247  int start = 0;
248  int end = m_pathsLengths.find (',');
249  while (end != -1)
250  {
251  m_numOnionPaths++;
252  start = end + 1; //increase by the delimiter size in this case one symbol (,)
253  end = m_pathsLengths.find (',', start);
254  }
255  m_onionPathsLengths = new uint16_t[++m_numOnionPaths];
256 
257  //insert the path length in single cells of the array m_onionPathsLengths
258  start = 0;
259  end = m_pathsLengths.find (',');
260  int i = 0;
261  while (end != -1)
262  {
263  m_onionPathsLengths[i] =
264  (uint16_t) (unsigned int) std::stoi (m_pathsLengths.substr (start, end - start));
265  start = end + 1; //increase by the delimiter size in this case one symbol (,)
266  end = m_pathsLengths.find (',', start);
267  i++;
268  }
269  m_onionPathsLengths[i] =
270  (uint16_t) (unsigned int) std::stoi (m_pathsLengths.substr (start, end - start));
271 }

373 void
374 WsnConstructor::BuildDiscTopology ()
375 {
376  //allocate3 positions
377  NS_LOG_INFO ("--------------- Building disc topology");
378 
379  //calculate radius of the disc based on node density and communication range
380  // compute the radius of a circle that covers the sum of the areas of each node
381  int r_disc = m_radius * sqrt (m_numNodes);
382 
383  m_simulationDescription =
384  m_simulationDescription +
385  "Network topology: RANDOM DISC, node radius: " + std::to_string (m_radius) +
386  "m , disc topology radius: " + std::to_string (r_disc) +
387  "m. Sink node located at x:" + std::to_string ((int) r_disc) +
388  ",y:" + std::to_string ((int) r_disc) + " \n";
389 
390  //Random disc
391  ObjectFactory rndDisc;
392  rndDisc.SetTypeId ("ns3::RandomDiscPositionAllocator");
393  rndDisc.Set ("Rho", StringValue ("ns3::UniformRandomVariable[Min=40|Max=" +
394  std::to_string (r_disc) + "]"));
395  rndDisc.Set ("X", DoubleValue (r_disc));
396  rndDisc.Set ("Y", DoubleValue (r_disc));
397  rndDisc.Set ("Z", DoubleValue (0));
398 
399  Ptr<PositionAllocator> posAlloc = rndDisc.Create ()->GetObject<PositionAllocator> ();
400 
401  mobility.SetPositionAllocator (posAlloc);
402  mobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
403  mobility.Install (sensornodes);
404 
405  Ptr<ListPositionAllocator> positionAlloc = CreateObject<ListPositionAllocator> ();
406  positionAlloc->Add (Vector (r_disc, r_disc, 0.0)); // only one position for sink
407  mobility.SetPositionAllocator (positionAlloc);
408  mobility.Install (m_sink);
409 }

411 /*
412 * allocate nodes on physical positions in space
413 *
414 * construct the grid topology
415 * sensor nodes are deployed in a plane following a grid structure; each sensor node is equidistant from the closest sensor nodes in cardinal directions
416 */

418 void
419 WsnConstructor::BuildGridTopology ()
420 {
421  NS_LOG_INFO ("--------------- Building grid topology");
422 
423  //calculate row size based on node number
424  int row_size = sqrt (m_numNodes);
425  int deltaX = m_cellSide;
426  int deltaY = m_cellSide;
427 
428  mobility.SetPositionAllocator ("ns3::GridPositionAllocator", "MinX", DoubleValue (0), "MinY",
429  DoubleValue (0), "DeltaX", DoubleValue (deltaX), "DeltaY",
430  DoubleValue (deltaY), "GridWidth", UintegerValue (row_size),
431  "LayoutType", StringValue ("RowFirst"));
432 
433  mobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
434  //put the sink around the middle of the grid
435  mobility.Install (sensornodes);
436  mobility.Install (m_sink);
437 
438  //change the location of the sink node to center by swap it with a node in the center
439  Ptr<MobilityModel> sinkMobility = m_sink.Get (0)->GetObject<MobilityModel> ();
440 
441  //Get the middle sensor node
442  int m = row_size / 2 * row_size + row_size / 2;
443 
444  Ptr<MobilityModel> middleNodeMobility = sensornodes.Get (m)->GetObject<MobilityModel> ();
445  double middleNode_x = middleNodeMobility->GetPosition ().x;
446  double middleNode_y = middleNodeMobility->GetPosition ().y;
447  //swap node positions
448  middleNodeMobility->SetPosition (
449  Vector (sinkMobility->GetPosition ().x, sinkMobility->GetPosition ().y, 0));
450  sinkMobility->SetPosition (Vector (middleNode_x, middleNode_y, 0));
451 
452  m_simulationDescription = m_simulationDescription +
453  "Network topology: GRID with row size: " + std::to_string (row_size) +
454  ". Distance between nodes on X-axis: " + std::to_string (deltaX) +
455  "m. Distance between nodes on Y-axis: " + std::to_string (deltaY) +
456  "m. Sink node located at x:" + std::to_string ((int) middleNode_x) +
457  ",y:" + std::to_string ((int) middleNode_y) + " \n";
458 }

570 void
571 WsnConstructor::InstallApplications ()
572 {
573  NS_LOG_INFO ("--------------- Configuring Applications ");
574 
575  //olsr is a proactive routing algo and need some time to compute routes
576  //at least 30s
577  int routing_setup_time = 20;
578 
579  if (m_routing == Routing::OLSR || m_routing == Routing::DSDV)
580  {
581  routing_setup_time = 60;
582  }
583 
584  //sink helper
585  SinkHelper msh (m_numNodes - 1, m_outputManager, m_onionValidator, m_onionPathsLengths);
586  //node helper - create nodes helpers to install node application
587  SensorNodeHelper mnh (wifiInterfaces.GetAddress (0), m_outputManager, m_onionValidator);
588 
589  //install sink apps
590  sinkApps = msh.Install (m_sink);
591  //install node apps
592  sensornodeApps.Add (mnh.Install (sensornodes));
593 
594  //setup onion routing settings on the sink node
595  sinkApps.Get (0)->GetObject<Sink> ()->Setup (m_onionPathsLengths, m_numOnionPaths,
596  m_onionRepeate);
597 
598  //start apps
599  sinkApps.Start (Seconds (1.0 + routing_setup_time));
600  sensornodeApps.Start (Seconds (2.0 + routing_setup_time));
601 
602  UintegerValue node_delay;
603  sinkApps.Get (0)->GetObject<Sink> ()->GetAttribute ("Delay", node_delay);
604  int start_onion = (node_delay.Get () * m_numNodes) / 1000 +
605  5; // when the sink node will start send onions in seconds
606 
607  m_simulationDescription =
608  m_simulationDescription + "Routing setup time: " + std::to_string (routing_setup_time) +
609  "s, nodes are starting sequentially with " + std::to_string (node_delay.Get ()) +
610  "ms interval, onion starts at: " + std::to_string (start_onion + routing_setup_time) + "s\n";
611 
612  m_simulationDescription = m_simulationDescription + "Onion path lengths: ";
613 
614  for (int i = 0; i < m_numOnionPaths; ++i)
615  {
616  m_simulationDescription =
617  m_simulationDescription + std::to_string (m_onionPathsLengths[i]) + " ";
618  }
619  m_simulationDescription = m_simulationDescription + " repeate each path lenght " +
620  std::to_string (m_onionRepeate) + " times.\n";
621 }
