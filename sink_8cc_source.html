94 void
95 Sink::ReceivePacket (Ptr<Socket> socket)
96 {
97  Address from;
98  Ptr<Packet> p = Wsn_node::RecvSegment (socket, from);
99 
100  if (p != NULL)
101  {
102 
103  InetSocketAddress address = InetSocketAddress::ConvertFrom (from);
104  SerializationWrapper sw;
105  p->RemoveHeader (sw);
106  protomessage::ProtoPacket message;
107  sw.GetData (&message);
108 
109  if (message.has_h_shake ())
110  {
111  //register node
112  RecvHandshake (message.mutable_h_shake (), address);
113  }
114  else //onion message
115  {
116  //get the onion ID
117  o_sequenceNum = message.mutable_o_head ()->onionid ();
118 
119  //
120  if (o_sequenceNum == m_onionId - 1)
121  { //message IDs are equal, -1 since m_onionId is larger for one value
122 
123  //call that onion was received
124  Wsn_node::OnionReceived ();
125 
126  RecvOnion (message.mutable_o_body ());
127  }
128  else
129  { //the onion should be deleted
130  NS_LOG_INFO ("Ghost onion received, deleted with onion id: "
131  << o_sequenceNum << ", at ip: " << m_address
132  << ", at time: " << std::to_string (Simulator::Now ().GetSeconds ()));
133  }
134  }

160 //defines the behaviour of the source how many OR sends, how to build the route ecc...
161 void
162 Sink::SinkTasks ()
163 {
164 
165  if (m_onionLengthIndex < m_numOnionLengths)
166  {
167 
168  if (m_repeateCount < m_repeateTimes)
169  {
170 
171  //selet the route of the onion
172  int route[m_onionPathLengths[m_onionLengthIndex]];
173  SelectRoute (route, m_onionPathLengths[m_onionLengthIndex]);
174  //send the onion
175  PrepareOnion (route, m_onionPathLengths[m_onionLengthIndex]);
176 
177  m_repeateCount++;
178  }
179  else
180  {
181  m_repeateCount = 0;
182  m_onionLengthIndex++;
183  SinkTasks (); // call again with new params
184  }
185  }
186  else
187  {
188  //simulation ended, can print details of nodes
189  m_outputManager->PrintNodeDetails (m_nodeManager);
190 
191  //end simulation
192  Simulator::Stop ();
193  }
194 }

200 void
201 Sink::SelectRoute (int *route, int routeLen)
202 {
203 
204  assert (routeLen >= 3); // the route must be of lentgh at least 3
205 
206  Ptr<UniformRandomVariable> x = CreateObject<UniformRandomVariable> ();
207  int node_id;
208 
209  //previous id -> previous and current must not be equal
210  int prev_id = -1;
211 
212  //build the route
213  int i = routeLen - 1;
214  while (i >= 0)
215  {
216  node_id = x->GetInteger (0, m_nodeManager.size () - 1);
217  if (node_id != prev_id)
218  {
219  route[i] = node_id;
220  i--;
221  }
222  prev_id = node_id;
223  }
224 }

228 void
229 Sink::PrepareOnion (int *route, int routeLen)
230 {
231  int cipherLen = m_onionManager.OnionLength (routeLen + 1, 0, 0);
232  unsigned char cipher[cipherLen];
233 
234  unsigned char *ipRoute[routeLen + 1];
235 
236  unsigned char *keys[routeLen + 1];
237 
238  //sink details
239  keys[routeLen] = m_onionManager.StringToUchar (m_publickey);
240  ipRoute[routeLen] =
241  m_onionManager.IpToBuff (m_address.Get ()); //set sink node as the last node in the onion path
242 
243  //define an iterator
244  std::map<uint32_t, std::string>::iterator item;
245 
246  //fill other addresses and publickeys
247  for (int i = 0; i < routeLen; ++i)
248  {
249  item = m_nodeManager.begin ();
250  std::advance (item, route[i]);
251 
252  ipRoute[i] = m_onionManager.IpToBuff (item->first);
253 
254  keys[i] = m_onionManager.StringToUchar (item->second);
255  }
256 
257  m_onionManager.BuildOnion (cipher, ipRoute, keys, routeLen + 1);
258 
259  item = m_nodeManager.begin ();
260  std::advance (item, route[0]);
261  SendOnion (item->first, routeLen, cipher, cipherLen);
262 }

264 void
265 Sink::SendOnion (uint32_t firstHop, int routeLen, unsigned char *cipher, int cipherLen)
266 {
267  //first convert cipher to string!!
268  std::string str_cipher = m_onionManager.UcharToString (cipher, cipherLen);
269  protomessage::ProtoPacket onion;
270 
271  //Create the onion head
272  onion.mutable_o_head ()->set_onion_message (str_cipher);
273  onion.mutable_o_head ()->set_onionid (m_onionId);
274 
275  // fixed onion head size
276  if (m_fixedOnionSize)
277  {
278  onion.mutable_o_head ()->set_padding ("");
279  }
280 
281  //Specify how the onion body must behave
282  std::string padding (m_bodySize, '0');
283  switch (m_bodyOptions)
284  {
285  case BodyOptions::NO_Body:
286  break;
287  case BodyOptions::Aggregate:
288  onion.mutable_o_body ()->set_aggregatedvalue (m_decoyNum);
289  break;
290  case BodyOptions::FixedSize:
291  onion.mutable_o_body ()->set_padding (padding);
292  break;
293  case BodyOptions::AggregateAndFixed:
294  onion.mutable_o_body ()->set_aggregatedvalue (m_decoyNum);
295  onion.mutable_o_body ()->set_padding (padding);
296  break;
297  }
298 
299  //Create the packet
300  SerializationWrapper sw (onion);
301  Ptr<Packet> p = Create<Packet> ();
302  p->AddHeader (sw);
303 
304  InetSocketAddress remote = InetSocketAddress (Ipv4Address (firstHop), m_port);
305 
306  Wsn_node::SendSegment (remote, p, true);
307 
308  m_onionValidator->StartOnion (m_onionId);
309  //increment onion sequence number
310  m_onionId++;
311 
312  m_outputManager->SendOnion (p->GetSize (), onion.mutable_o_head ()->ByteSizeLong (),
313  onion.mutable_o_body ()->ByteSizeLong (), routeLen,
314  Simulator::Now ());
315  m_outputManager->OnionRoutingSend (m_address, Ipv4Address (firstHop), p->GetSize (),
316  onion.mutable_o_head ()->ByteSizeLong (),
317  onion.mutable_o_body ()->ByteSizeLong (), Simulator::Now ());
318 }
