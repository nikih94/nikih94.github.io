78 void
79 Wsn_node::Configure ()
80 {
81  //Get the address of the node on which the application is running
82  Ptr<Node> PtrNode = this->GetNode ();
83  Ptr<Ipv4> ipv4 = PtrNode->GetObject<Ipv4> ();
84  Ipv4InterfaceAddress iaddr = ipv4->GetAddress (1, 0);
85  Ipv4Address address = iaddr.GetLocal ();
86  m_address = address;
87 
88  m_socket = Socket::CreateSocket (GetNode (), TcpSocketFactory::GetTypeId ());
89  InetSocketAddress local (Ipv4Address::GetAny (), m_port);
90  m_socket->SetIpRecvTtl (true);
91  m_socket->Bind (local);
92  m_socket->Listen ();
93 
94  //Cordinates of the node
95  Ptr<MobilityModel> mob = PtrNode->GetObject<MobilityModel> ();
96  double coord_x = mob->GetPosition ().x;
97  double coord_y = mob->GetPosition ().y;
98 
99  //if the routing is OLSR we can print out the number of one-hop neighbours from routing info
100  if (m_outputManager->GetRouting () == Routing::OLSR)
101  {
102  //need to wait some seconds till routes are stable
103  Simulator::Schedule (Seconds (5), &Wsn_node::NodeDegree, this, coord_x, coord_y);
104  }
105  else
106  {
107  m_outputManager->AddNodeDetails (m_address, coord_x, coord_y);
108  }
109 }

118 void
119 Wsn_node::NodeDegree (double coord_x, double coord_y)
120 {
121 
122  Ptr<Node> PtrNode = this->GetNode ();
124  Ptr<Ipv4> ipv4Ptr = PtrNode->GetObject<Ipv4> ();
125  Ptr<Ipv4RoutingProtocol> ipv4Routing = ipv4Ptr->GetRoutingProtocol ();
126  // Obtain the Ipv4ListRouting interface
127  Ptr<Ipv4ListRouting> ipv4ListRouting = DynamicCast<Ipv4ListRouting> (ipv4Routing);
128  // A list routing protocol has multiple Ipv4Routing protocols. Find OLSR in this list
129  Ptr<olsr::RoutingProtocol> olsrProtocol;
130  int16_t priority = 10;
131  for (uint32_t i = 0; i < ipv4ListRouting->GetNRoutingProtocols (); i++)
132  {
133  Ptr<Ipv4RoutingProtocol> proto = ipv4ListRouting->GetRoutingProtocol (0, priority);
134  olsrProtocol = DynamicCast<olsr::RoutingProtocol> (proto);
135  }
136  if (olsrProtocol != 0)
137  {
138  //break; // found the protocol we are looking for
139  }
140  else
141  {
142  NS_ASSERT_MSG (olsrProtocol, "Didn't find OLSR on this node");
143  }
144 
145  ns3::olsr::OlsrState state = olsrProtocol->GetOlsrState ();
146  ns3::olsr::NeighborSet n_set = state.GetNeighbors ();
147 
148  m_outputManager->AddNodeDetails (m_address, coord_x, coord_y, n_set.size ());
149 }

172 void
173 Wsn_node::SendSegment (InetSocketAddress remote, Ptr<Packet> packet, bool b_onion)
174 {
175 
176  Ptr<Socket> socket = Socket::CreateSocket (GetNode (), TcpSocketFactory::GetTypeId ());
177  socket->Connect (remote);
178 
179  int pack_size = packet->GetSize ();
180  int seg_num = pack_size / f_mss;
181 
182  if (b_onion)
183  {
184  this->o_hopCount = m_onionValidator->OnionHopCount ();
185  Simulator::Schedule (Seconds (m_onionTimeout), &Wsn_node::CheckSentOnion, this,
186  this->o_hopCount);
187  }
188 
189  if (seg_num == 0)
190  { //packet fits in segment size, just send-it
191  socket->Send (packet);
192  }
193  else
194  {
195 
196  SegmentNum s_num (pack_size);
197  packet->AddByteTag (s_num);
198  socket->Send (packet);
199  }
200 }

226 /*
227 * Receive a segment
228 * The tag represents the size of the whole packet to be received
229 * merge parts into the whole packet
230 * pending_packet is the buffer where you aggregate segments
231 *
232 */
233 
234 Ptr<Packet>
235 Wsn_node::RecvSeg (Ptr<Socket> socket, Ptr<Packet> p, Address from)
236 {
237 
238  InetSocketAddress from_address = InetSocketAddress::ConvertFrom (from);
239 
240  SegmentNum s_num;
241  //if only one packet
242  if (!p->FindFirstMatchingByteTag (s_num))
243  {
244  //tag not found
245  socket->Close ();
246  return p;
247  }
248 
249  if (f_pendingPacket == NULL || from_address.GetIpv4 ().Get () != f_receivingAddress.Get ())
250  {
251  f_receivingAddress = from_address.GetIpv4 (); //Ipv4Address(0);
252  f_pendingPacket = Create<Packet> ();
253  f_segmentSize = s_num.GetSegNum ();
254  }
255  f_pendingPacket->AddAtEnd (p);
256 
257  f_segmentSize = f_segmentSize - p->GetSize ();
258 
259  if (f_segmentSize == 0)
260  {
261  p = f_pendingPacket->Copy ();
262  f_receivingAddress = Ipv4Address::GetAny ();
263  f_pendingPacket = NULL;
264  socket->Close ();
265  return p;
266  }
267  else
268  {
269  return NULL;
270  }
271 }
